<script>
  const lines = [];

  function money(n) {
    const x = Number(n || 0);
    return x.toLocaleString("en-ZA", { minimumFractionDigits: 2, maximumFractionDigits: 2 });
  }

  function pct(n) {
    const x = Number(n || 0);
    return `${x.toFixed(2)}%`;
  }

  function clamp(x, min, max) {
    return Math.min(Math.max(x, min), max);
  }

  function applyDefaults() {
    const type = document.getElementById("propertyType").value;
    // sensible defaults (you can edit these)
    const defaults = {
      office: 88,
      retail: 92,
      industrial: 95,
      storage: 91,
      other: 90
    };
    document.getElementById("stabilisedOccPct").value = defaults[type] ?? 90;
  }

  function addLine() {
    const desc = document.getElementById("desc").value.trim() || "Rentable Area";
    const size = Number(document.getElementById("size").value || 0);
    const rate = Number(document.getElementById("rate").value || 0);

    if (size <= 0 || rate < 0) return;

    lines.push({ desc, size, rate });
    document.getElementById("desc").value = "";
    document.getElementById("size").value = "";
    document.getElementById("rate").value = "";
    render();
    recalc();
  }

  function removeLine(i) {
    lines.splice(i, 1);
    render();
    recalc();
  }

  function render() {
    const tbody = document.querySelector("#linesTable tbody");
    tbody.innerHTML = "";
    lines.forEach((l, i) => {
      const monthly = l.size * l.rate;
      const tr = document.createElement("tr");
      tr.innerHTML = `
        <td>${l.desc}</td>
        <td>${money(l.size)}</td>
        <td>${money(l.rate)}</td>
        <td>${money(monthly)}</td>
        <td><button onclick="removeLine(${i})">Remove</button></td>
      `;
      tbody.appendChild(tr);
    });
  }

  function recalc() {
    const otherMonthly = Number(document.getElementById("otherMonthly").value || 0);

    const pgiMonthly =
      lines.reduce((sum, l) => sum + (l.size * l.rate), 0) + otherMonthly;

    const pgiAnnual = pgiMonthly * 12;

    document.getElementById("pgiMonthlyOut").textContent = `R ${money(pgiMonthly)}`;
    document.getElementById("pgiAnnualOut").textContent = `R ${money(pgiAnnual)}`;

    // Actual annual revenue & derived occupancy
    const actualAnnualRev = Number(document.getElementById("actualAnnualRev").value || 0);
    let derivedOcc = null;

    if (pgiAnnual > 0 && actualAnnualRev > 0) {
      derivedOcc = clamp((actualAnnualRev / pgiAnnual) * 100, 0, 200); // allow >100 if data is messy
      document.getElementById("currentOccOut").textContent = pct(derivedOcc);
      document.getElementById("currentVacOut").textContent = pct(clamp(100 - derivedOcc, 0, 100));
    } else {
      document.getElementById("currentOccOut").textContent = "—";
      document.getElementById("currentVacOut").textContent = "—";
    }

    // Assumptions
    const stabilisedOccPct = clamp(Number(document.getElementById("stabilisedOccPct").value || 0), 0, 100);
    const opexAnnual = Number(document.getElementById("opexAnnual").value || 0);
    const utilityAdj = Number(document.getElementById("utilityAdj").value || 0);

    const capLow = Number(document.getElementById("capLowPct").value || 0) / 100;
    const capHigh = Number(document.getElementById("capHighPct").value || 0) / 100;

    // swap if user entered backwards
    let capMin = Math.min(capLow, capHigh);
    let capMax = Math.max(capLow, capHigh);

    const excessLand = Number(document.getElementById("excessLand").value || 0);
    const refurb = Number(document.getElementById("refurb").value || 0);

    // Scenario selection
    const scenario = document.getElementById("scenario").value;

    // Compute EGI used
    let egiUsed = 0;
    let valueNote = "";
    let showActualNote = false;

    // warning about double vacancy (informational)
    const warnEl = document.getElementById("doubleVacancyWarning");
    warnEl.style.display = "none";

    if (scenario === "actual") {
      egiUsed = actualAnnualRev; // by definition: already includes vacancy/collection reality
      valueNote = "Using actual annual revenue (no additional vacancy applied).";
      showActualNote = true;

      if (actualAnnualRev > 0 && pgiAnnual > 0 && actualAnnualRev < pgiAnnual) {
        warnEl.style.display = "block";
      }
    } else {
      // stabilised
      egiUsed = pgiAnnual * (stabilisedOccPct / 100);
      valueNote = `Using stabilised income (PGI × ${pct(stabilisedOccPct)}).`;
      showActualNote = false;
    }

    document.getElementById("egiUsedOut").textContent = `R ${money(egiUsed)}`;

    // NOI
    const noi = egiUsed - opexAnnual + utilityAdj;
    document.getElementById("noiOut").textContent = `R ${money(noi)}`;

    // Value range (cap higher => lower value)
    let vLow = 0, vHigh = 0;
    if (capMin > 0 && capMax > 0) {
      // vHigh uses lower cap; vLow uses higher cap
      vHigh = (noi / capMin) + excessLand - refurb;
      vLow = (noi / capMax) + excessLand - refurb;
    }

    // format
    const lo = Math.min(vLow, vHigh);
    const hi = Math.max(vLow, vHigh);

    document.getElementById("valueRangeOut").textContent =
      `Value Range: R ${money(lo)} – R ${money(hi)}`;

    document.getElementById("valueNoteOut").textContent = valueNote;

    const actualNoteEl = document.getElementById("actualScenarioNote");
    actualNoteEl.style.display = showActualNote ? "block" : "none";
  }

  // Recalc on typing for convenience
  ["otherMonthly","actualAnnualRev","stabilisedOccPct","opexAnnual","utilityAdj","capLowPct","capHighPct","excessLand","refurb"]
    .forEach(id => document.getElementById(id).addEventListener("input", recalc));

  applyDefaults();
  recalc();
</script>
